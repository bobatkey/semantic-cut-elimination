Negation:
\begin{displaymath}
  \begin{array}{lcl}
    \Dual(\Pos(A))   & = & \Neg(A)                 \\
    \Dual(\Neg(A))   & = & \Pos(A)                 \\
    \Dual(\Unit)     & = & \Unit                   \\
    \Dual(P \Tens Q) & = & \Dual(P) \Parr \Dual(Q) \\
    \Dual(P \Parr Q) & = & \Dual(P) \Tens \Dual(Q) \\
    \Dual(P \With Q) & = & \Dual(P) \Plus \Dual(Q) \\
    \Dual(P \Plus Q) & = & \Dual(P) \With \Dual(Q) \\
    \Dual(P \Seq  Q) & = & \Dual(P) \Seq  \Dual(Q)
  \end{array}
\end{displaymath}
